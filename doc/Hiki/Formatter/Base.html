def apply_tdiary_theme(orig_html)
  return orig_html if @conf.mobile_agent?
  section = ""
  title   = ""
  html    = ""

  orig_html.each_line do |line|
    if H2_RE =~ line
      html << tdiary_section(title, section) unless title.empty? && section.empty?
      section = ""
      title = line
    else
      section << line
    end
  end
  html << tdiary_section(title, section)
end
