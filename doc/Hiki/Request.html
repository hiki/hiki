def cookies
  return @cookies if @cookies
  @cookies = { }
  @cgi.cookies.each{|k, v|
    case v.size
    when 0
      @cookies[k] = nil
    when 1
      @cookies[k] = v[0]
    else
      @cookies[k] = v
    end
  }
  @cookies
end

def params
  return @params if @params
  @params = { }
  @cgi.params.each{|k, v|
    v = v.uniq
    case v.size
    when 0
      @params[k] = nil
    when 1
      @params[k] = v[0]
    else
      @params[k] = v
    end
  }
  @params
end
