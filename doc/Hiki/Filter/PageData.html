def initialize(page=nil, text=nil, title=nil, keyword=nil, remote_addr=nil)
  @page = page
  @text = text
  @title = title==page ? nil : title
  keyword = keyword.split("\n") if keyword.is_a?(String)
  @keyword = keyword || []
  @remote_addr = remote_addr
end
