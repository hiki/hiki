def pages
  entries = Dir.entries(File.join(@data_path, "text")).reject{|entry|
    entry =~ /\A\./ || File.directory?(entry)
  }.map{|entry| unescape(entry).b }
  if block_given?
    entries.each do |entry|
      yield entry
    end
  else
    entries
  end
end
