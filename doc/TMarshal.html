  end
  dumped
end

def dump_text(obj)
  case obj
  when String
    obj.dump
  when Array
    "[\n"+obj.collect{|x| dump_text(x)+",\n"}.join+"]"
  when Hash
    "{\n"+obj.sort_by{|e| e[0].inspect}.collect{|k,v| "#{dump_text(k)} => #{dump_text(v)},\n"}.join+"}"
  when Numeric, Module, Regexp, Symbol, TrueClass, FalseClass, NilClass, Range
    obj.inspect
  when Time
    "Time.at(#{obj.to_i})"
  else
    raise "Wrong type!"
  end
end

def load(port)
  case port
  when String
    eval port.untaint
  when IO, StringIO
    eval port.read.untaint
  else
    raise "Wrong type!"
  end
end
